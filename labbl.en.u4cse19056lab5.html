class Perceptron:
    def __init__(self,weights,alfa,epochs):
        self.weights = weights
        self.learning_rate = alfa
        self.epochs = epochs
        self.perror = []
    def train(self,data):
        print(f"Intial weig: {self.weights}")
        for i in range(self.epochs):
            total_error = 0
            for j in data:
                yin = j[0]*self.weights[0]+j[1]*self.weights[1]+self.weights[2]
                if(yin >0):
                    y=1
                elif(yin == 0):
                    y=0
                else:
                    y=-1
                error = (j[2]-y)
                total_error += error**2
                self.weights[0] += self.learning_rate*error*j[0]
                self.weights[1] += self.learning_rate*error*j[1]
                self.weights[2] += self.learning_rate*error
            self.perror.append(total_error)
        print(f"Final Wei: {self.weights} err: {total_error} ")
        return self.perror

class Perceptron:
    def __init__(self,weights,alfa,epochs,activation_function):
        self.weights = weights
        self.learning_rate = alfa
        self.activation_function = activation_function
        self.epochs = epochs
        self.perror = []
    def train(self,data):
        
        print(f"Initial Weights: {self.weights}")
        for i in range(self.epochs):
            total_error = 0
            for j in data:
                y = j[0]*self.weights[0]+j[1]*self.weights[1]+self.weights[2]
                if self.activation_function == "relu":
                    z = self.relu(y)
                elif self.activation_function == "sigmoid":
                    z = self.sigmoid(y)
                elif self.activation_function == "tanh":
                    z = self.tanh(y)
                elif self.activation_function == "sign":
                    z = self.sign(y)
                error = (j[2]-y)
                total_error += error**2
                self.weights[0] += self.learning_rate*error*j[0]
                self.weights[1] += self.learning_rate*error*j[1]
                self.weights[2] += self.learning_rate*error
            self.perror.append(total_error)
        print(f"Final Weights: {self.weights} Final cost: {total_error} ")
        return self.perror
    def relu(self,x):
        if x>=0: return 1
        else: return 0 
    
    def sigmoid(self,z):
        return 1/(1+np.exp(-z))
    
    def tanh(self,z):
        p = np.exp(z)
        n = np.exp(-z)
        return (p-n)/(p+n)
    
    def sign(self,y):
        if y>0: return 1
        elif y==0: return 0
        else: return -1

class Perceptron:
    
    def __init__(self,weights,learning_rate,epochs,activation_function):
        
        self.weights = weights
        self.learning_rate = learning_rate
        self.epochs = epochs
        self.activation_function = activation_function
        self.perror = []
        
    def train(self,data):
        
        print(f"Initial Weights: {self.weights}")
        for i in range(self.epochs):
            total_error = 0
            for j in data:
                y=j[0]*self.weights[0]+j[1]*self.weights[1]+j[2]*self.weights[2]+j[3]*self.weights[3]+self.weights[4]
                if self.activation_function == "relu":
                    z = self.relu(y)
                elif self.activation_function == "sigmoid":
                    z = self.sigmoid(y)
                elif self.activation_function == "tanh":
                    z = self.tanh(y)
                elif self.activation_function == "sign":
                    z = self.sign(y)
                error = (j[2]-z)
                total_error += error**2
                self.weights[0] += self.learning_rate*error*j[0]
                self.weights[1] += self.learning_rate*error*j[1]
                self.weights[2] += self.learning_rate*error*j[2]
                self.weights[3] += self.learning_rate*error*j[3]
                self.weights[4] += self.learning_rate*error
            self.perror.append(total_error)
        print(f"Final Weights: {self.weights} Final cost: {total_error} ")
        return self.perror
                
    
    def relu(self,x):
        if x>=0: return 1
        else: return 0 
    
    def sigmoid(self,z):
        return 1/(1+np.exp(-z))
    
    def tanh(self,z):
        p = np.exp(z)
        n = np.exp(-z)
        return (p-n)/(p+n)
    
    def sign(self,y):
        if y>0: return 1
        elif y==0: return 0
        else: return -1
    def plot_cost(self,data_title):
        plt.figure(figsize=(10,6))
        plt.plot(range(self.epochs),self.perror)
        plt.ylabel("Cost")
        plt.xlabel("Epochs")
        plt.title(f"Epochs vs Cost, AC={self.activation_function} Alpha={self.learning_rate} data={data_title}")
        plt.show()
    def Accuracy(self,data):
        pred=[]
        for j in data:
            y=j[0]*self.weights[0]+j[1]*self.weights[1]+j[2]*self.weights[2]+j[3]*self.weights[3]+self.weights[4]
            if self.activation_function=="relu":
                z=self.relu(y)
                pred.append(z)
            elif self.activation_function=="sigmoid":
                z=self.sigmoid(y)
                if z>=0.5:
                    pred.append(1)
                else:
                    pred.append(0)
            elif self.activation_function=="tanh":
                z=self.tanh(y)
                if z>=0:
                    pred.append(1)
                else:
                    pred.append(0)
        original=[i[-1] for i in data]
        accuracy=(pred==np.squeeze(original)).sum()/len(original)
        print(f"Original labels : {original}")
        print(f"Predicted labels : {pred}")
        print(f"Accuracy is {accuracy*100}%")
